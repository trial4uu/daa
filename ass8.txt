PART B — HIVE (Data Mining)

✅ Step 7: Open Hive
hive

✅ Step 8: Create Table
CREATE TABLE forest (
temp INT,
humidity INT,
wind INT,
area FLOAT
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ',';

✅ Step 9: Load Data
LOAD DATA INPATH '/forest/forest.csv'
INTO TABLE forest;

✅ Step 10: Query
SELECT temp, AVG(area)
FROM forest
GROUP BY temp;

📊 PART C — VISUALIZATION (Python)

✅ Step 11: Python Code
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv("forest.csv")

# Line Plot
plt.plot(df['temp'], df['area'])
plt.title("Temp vs Area")
plt.show()

# Bar Plot
sns.barplot(x='temp', y='area', data=df)
plt.show()

# Heatmap
sns.heatmap(df.corr(), annot=True)
plt.show()
