/*Assignment 2(B)
Question:
Implement the C program in which main program accepts an array. Main program uses the FORK system call to create a new process called a child process. Parent process sorts an array and passes the sorted array to child process through the command line arguments of EXECVE system call. The child process uses EXECVE system call to load new program that display array in reverse order.
*/
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>

// Bubble sort function
void bubbleSort(int arr[], int n) {
    int i, j, temp;

    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - i - 1; j++) {

            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {

    int n, i;

    printf("Enter the number of integers: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter %d integers:\n", n);

    for (i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    pid_t pid = fork();

    if (pid < 0) {
        perror("Fork failed");
        exit(1);
    }

    // Child Process
    else if (pid == 0) {

        printf("\nChild Process (PID: %d)\n", getpid());

        printf("Displaying integers in reverse order:\n");

        for (i = n - 1; i >= 0; i--)
            printf("%d ", arr[i]);

        printf("\nChild process completed.\n");

        exit(0);
    }

    // Parent Process
    else {

        printf("\nParent Process (PID: %d)\n", getpid());

        bubbleSort(arr, n);

        printf("Parent sorting integers in ascending order:\n");

        for (i = 0; i < n; i++)
            printf("%d ", arr[i]);

        printf("\n");

        // Parent waits for child
        wait(NULL);

        printf("Child process finished. Parent exiting.\n");
    }

    return 0;
}
