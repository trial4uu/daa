#include <stdio.h>

int main()
{
    int pages[50], frames[10];
    int n, f, i, j, k;

    printf("Enter number of pages: ");
    scanf("%d", &n);

    printf("Enter page reference string:\n");

    for(i = 0; i < n; i++)
        scanf("%d", &pages[i]);

    printf("Enter number of frames: ");
    scanf("%d", &f);

    // ================= FIFO =================

    int pos = 0, fault = 0, found;
    printf("\n----- FIFO -----\n");

    for(i = 0; i < f; i++)
        frames[i] = -1;

    for(i = 0; i < n; i++){
        found = 0;

        for(j = 0; j < f; j++){
            if(frames[j] == pages[i]){
                found = 1;
                break;
            }
        }
        if(found == 0){
            frames[pos] = pages[i];
            pos = (pos + 1) % f;
            fault++;
        }

        printf("\nFrames: ");

        for(j = 0; j < f; j++)
            printf("%d ", frames[j]);
    }

    printf("\nTotal Page Faults = %d\n", fault);

    // ================= LRU =================

    int time[10], least, min, counter = 0;
    fault = 0;
    printf("\n----- LRU -----\n");

    for(i = 0; i < f; i++){
        frames[i] = -1;
        time[i] = 0;
    }

    for(i = 0; i < n; i++){
        found = 0;
        for(j = 0; j < f; j++){
            if(frames[j] == pages[i]){
                counter++;
                time[j] = counter;
                found = 1;
                break;
            }
        }

        if(found == 0){
            min = time[0];
            least = 0;

            for(j = 1; j < f; j++){
                if(time[j] < min){
                    min = time[j];
                    least = j;
                }
            }

            counter++;
            frames[least] = pages[i];
            time[least] = counter;
            fault++;
        }

        printf("\nFrames: ");

        for(j = 0; j < f; j++)
            printf("%d ", frames[j]);
    }

    printf("\nTotal Page Faults = %d\n", fault);

    // ================= OPTIMAL =================

    int farthest;
    fault = 0;
    printf("\n----- OPTIMAL -----\n");

    for(i = 0; i < f; i++)
        frames[i] = -1;

    for(i = 0; i < n; i++){
        found = 0;
        for(j = 0; j < f; j++){
            if(frames[j] == pages[i]){
                found = 1;
                break;
            }
        }

        if(found == 0){
            pos = -1;
            farthest = i + 1;

            for(j = 0; j < f; j++){
                for(k = i + 1; k < n; k++){
                    if(frames[j] == pages[k]){
                        if(k > farthest){
                            farthest = k;
                            pos = j;
                        }
                        break;
                    }
                }

                if(k == n){
                    pos = j;
                    break;
                }
            }
            if(pos == -1)
                pos = 0;

            frames[pos] = pages[i];
            fault++;
        }

        printf("\nFrames: ");

        for(j = 0; j < f; j++)
            printf("%d ", frames[j]);
    }

    printf("\nTotal Page Faults = %d\n", fault);
    return 0;
}
